## Terminal Raw Mode

Terminals normally operate in **cooked mode**: input is line-buffered and special keys are interpreted. For an editor, we need **raw mode**: every keypress delivered immediately.

## The Gap Buffer

A gap buffer is a simple data structure for text editing. It's an array with a "gap" at the cursor position:

<Editor defaultCode={`#include <stdio.h>
#include <string.h>

#define BUF_SIZE 256

typedef struct {
    char buf[BUF_SIZE];
    int gap_start;  // Start of gap
    int gap_end;    // End of gap (exclusive)
    int size;       // Total buffer size
} GapBuffer;

void gb_init(GapBuffer *gb) {
    memset(gb->buf, 0, BUF_SIZE);
    gb->gap_start = 0;
    gb->gap_end = BUF_SIZE;
    gb->size = BUF_SIZE;
}

int gb_length(GapBuffer *gb) {
    return gb->size - (gb->gap_end - gb->gap_start);
}

void gb_insert(GapBuffer *gb, char c) {
    if (gb->gap_start == gb->gap_end) return;  // Buffer full
    gb->buf[gb->gap_start++] = c;
}

void gb_delete(GapBuffer *gb) {
    if (gb->gap_start > 0) {
        gb->gap_start--;
    }
}

void gb_move_left(GapBuffer *gb) {
    if (gb->gap_start > 0) {
        gb->gap_end--;
        gb->buf[gb->gap_end] = gb->buf[gb->gap_start - 1];
        gb->gap_start--;
    }
}

void gb_move_right(GapBuffer *gb) {
    if (gb->gap_end < gb->size) {
        gb->buf[gb->gap_start] = gb->buf[gb->gap_end];
        gb->gap_start++;
        gb->gap_end++;
    }
}

// Get the text content (without gap)
void gb_contents(GapBuffer *gb, char *out) {
    int j = 0;
    for (int i = 0; i < gb->gap_start; i++) out[j++] = gb->buf[i];
    for (int i = gb->gap_end; i < gb->size; i++) out[j++] = gb->buf[i];
    out[j] = '\\0';
}

// Visualize the buffer internals
void gb_debug(GapBuffer *gb) {
    printf("Buffer: [");
    for (int i = 0; i < gb->gap_start; i++) printf("%c", gb->buf[i]);
    printf("|GAP(%d)|", gb->gap_end - gb->gap_start);
    for (int i = gb->gap_end; i < gb->size; i++) printf("%c", gb->buf[i]);
    printf("]\\n");
}

int main() {
    GapBuffer gb;
    gb_init(&gb);

    printf("=== Gap Buffer Demo ===\\n\\n");

    // Type "Hello World"
    const char *text = "Hello World";
    for (int i = 0; text[i]; i++) gb_insert(&gb, text[i]);
    gb_debug(&gb);

    char out[BUF_SIZE];
    gb_contents(&gb, out);
    printf("Text: '%s' (len=%d)\\n\\n", out, gb_length(&gb));

    // Move cursor left 5 times (past "World")
    for (int i = 0; i < 5; i++) gb_move_left(&gb);
    printf("After moving left 5:\\n");
    gb_debug(&gb);

    // Delete one char (the space)
    gb_delete(&gb);
    printf("After deleting 1 char:\\n");
    gb_debug(&gb);

    // Insert ", "
    gb_insert(&gb, ',');
    gb_insert(&gb, ' ');
    printf("After inserting ', ':\\n");
    gb_debug(&gb);

    gb_contents(&gb, out);
    printf("\\nFinal text: '%s'\\n", out);

    return 0;
}
`} />

## Editor Architecture

<Editor defaultCode={`#include <stdio.h>
#include <string.h>

// A minimal editor has these components:

// 1. ROWS: array of text lines
#define MAX_ROWS 1000
#define MAX_COLS 256

typedef struct {
    char chars[MAX_COLS];
    int len;
} Row;

typedef struct {
    Row rows[MAX_ROWS];
    int num_rows;
    int cursor_x, cursor_y;
    int row_offset;     // Scroll position
    int col_offset;
    int screen_rows;
    int screen_cols;
    char filename[256];
    int modified;
} Editor;

void editor_init(Editor *e) {
    memset(e, 0, sizeof(Editor));
    e->screen_rows = 24;
    e->screen_cols = 80;
}

void editor_insert_char(Editor *e, char c) {
    if (e->cursor_y >= e->num_rows) {
        // Add new row
        e->rows[e->num_rows].len = 0;
        e->num_rows++;
    }

    Row *row = &e->rows[e->cursor_y];
    if (row->len < MAX_COLS - 1) {
        // Shift chars right
        memmove(&row->chars[e->cursor_x + 1],
                &row->chars[e->cursor_x],
                row->len - e->cursor_x);
        row->chars[e->cursor_x] = c;
        row->len++;
        e->cursor_x++;
        e->modified = 1;
    }
}

void editor_insert_newline(Editor *e) {
    if (e->num_rows >= MAX_ROWS - 1) return;

    Row *cur = &e->rows[e->cursor_y];

    // Shift rows down
    for (int i = e->num_rows; i > e->cursor_y + 1; i--) {
        e->rows[i] = e->rows[i - 1];
    }
    e->num_rows++;

    // Split current row
    Row *next = &e->rows[e->cursor_y + 1];
    int split = e->cursor_x;
    memcpy(next->chars, &cur->chars[split], cur->len - split);
    next->len = cur->len - split;
    cur->len = split;

    e->cursor_y++;
    e->cursor_x = 0;
    e->modified = 1;
}

int main() {
    Editor e;
    editor_init(&e);

    // Simulate typing
    const char *lines[] = {
        "#include <stdio.h>",
        "",
        "int main() {",
        "    printf(\"Hello!\\\\n\");",
        "    return 0;",
        "}"
    };

    for (int i = 0; i < 6; i++) {
        for (int j = 0; lines[i][j]; j++) {
            editor_insert_char(&e, lines[i][j]);
        }
        editor_insert_newline(&e);
    }

    printf("Editor state: %d rows\\n\\n", e.num_rows);
    for (int i = 0; i < e.num_rows; i++) {
        e.rows[i].chars[e.rows[i].len] = '\\0';
        printf("%3d | %s\\n", i + 1, e.rows[i].chars);
    }

    printf("\\nCursor at: row %d, col %d\\n", e.cursor_y + 1, e.cursor_x);
    printf("Modified: %s\\n", e.modified ? "yes" : "no");

    return 0;
}
`} />
